Tree

UI component which can be used when displaying data in a hierarchical structure, as in an explorer 

          <ul class="tree_1 tree tree-arrow"></ul>
<ul class="tree_1 tree tree-line"></ul>
<ul class="tree_1 tree tree-arrow-file"></ul>
<ul class="tree_1 tree tree-line-file"></ul>
          <script id="tpl_tree" type="text/template">
	<li>
		<div><i></i> <!= title !></div>
		<ul></ul>
	</li>
</script>
          jui.ready([ "uix.tree" ], function(tree) {
	tree_1 = tree(".tree", {
		root: { title: "C:\\" },
		event: {
			select: function(node) {
				this.select(node.index); // selection effect
				alert("index(" + node.index + "), title(" + node.data.title + ")");
			}
		},
		tpl: {
			node: $("#tpl_tree").html()
		}
	});
});

function addTreeData(tree) {
	tree.append({ title: "Windows" });
	tree.append({ title: "Download" });
	tree.append({ title: "Program Files" });
	tree.append({ title: "Apache" });
	tree.append("0", { title: "run.exe" });
	tree.append("0", { title: "setting.conf" });
	tree.append("1", { title: "jui.torrrent" });
	tree.insert("2.0", { title: "Riot Games" });
	tree.insert("2.0.0", { title: "lol.exe" });
	tree.append("3", { title: "startup.bat" });
	tree.fold("0");
	tree.fold("1");
	tree.fold("3");
}

for(var i = 0; i < tree_1.length; i++) {
	addTreeData(tree_1[i]);
}

          Move/Remove/Update Tree

          Move Remove Update - Provides methods to control a node. 

            function tree2RunA() {
	alert("The Apache folder will be moved to the Program Files folder.");
	tree_1[4].move("3", "2.1");
}

function tree2RunB() {
	alert("The Download folder will be deleted.");
	tree_1[4].remove("1");
}
			
function tree2RunC() {
	alert("The Program Files folder will be renamed.");
	tree_1[4].update("2", { title: "Applications" });
}

            Move Node (+Drag)

            Provides a method for drag each node with a mouse. 

              tree_2 = tree("#tree_2", {ging
	root: { title: "C:\\" },
	drag: true,
	tpl: {
		node: $("#tpl_tree").html()
	}
});

              Move Node (+Drag Child)

              It is possible to prevent a target node being dragged from changing to a child of an opponent node if the dragChild option value is set to false. 

                tree_3 = tree("#tree_3", {
	root: { title: "C:\\" },
	drag: true,
	dragChild: false,
	tpl: {
		node: $("#tpl_tree").html()
	}
});
