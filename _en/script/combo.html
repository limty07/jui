The following is a code to create a check button. 

<div id="combo_1" class="combo">
	<a class="btn btn-gray">Select...</a>
	<a class="btn btn-gray btn-toggle"><i class="icon-arrow2"></i></a>
	<ul>
		<li value="1">New</li>
		<li value="2">Open</li>
		<li value="3">Save</li>
		<li value="4">Close</li>
		<li value="5">Restart</li>
		<li class="divider"></li>
		<li value="6">Print</li>
		<li class="divider"></li>
		<li value="7">Exit</li>
	</ul>
</div>
jui.ready([ "ui.combo" ], function(combo) {
	combo_1 = combo("#combo_1", {
		index: 2,
		position: "top",
		event: {
			change: function(data) {
				alert("text(" + data.text + "), value(" + data.value + ")");
			}
		}
	});
	
	combo_2 = combo("#combo_2", {
		index: 3,
		width: 196,
		keydown: true,
		event: {
			change: function(data) {
				alert("text(" + data.text + "), value(" + data.value + ")");
			}
		}
	});
});
