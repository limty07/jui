The user can select a total of 3 options. count is the total number of data subjects in a paging process.
pageCount is the number of data per page. screenCount is the number of pages shown in the paging UI. 

jui.ready([ "ui.paging" ], function(paging) {
	paging_1 = paging("#paging_1", {
		count: 1000,
		pageCount: 10,
		screenCount: 7, 
		event: {
			page: function(pNo) {
				alert(pNo);
			}
		}
	});
});

A custom page event occurs when you click on the page button, and at that time, the page shown in the table changes. 

paging_2 = paging("#paging_2", {
	pageCount: 100, 
	event: {
		page: function(pNo) {
			paging_2_xtable.page(pNo);
		}
	}
});

If the total number of rows is changed as the table is updated, the user just needs to call the reload method. 

function paging2Run() {
	xtableRun();
	paging_2.reload(paging_2_xtable.count());
}

When using the sort functionality of the table, calling the first method returns the page number to the starting page number. 

paging_2_xtable = xtable("#paging_2_xtable", {
	fields: [ "name", "age", "location" ],
	resize: true,
	sort: true,
	sortLoading: true,
	buffer: "s-page", // scroll, page, s-page
	bufferCount: 100,
	event: {
		sortend: function(data, e) {
			paging_2.first();
		}
	}
});
