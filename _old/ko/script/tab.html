Manipulation

Append Prepend Insert Remove Move - node 템플릿이 설정되어 있다면 좌측 버튼 기능을 사용할 수 있습니다.

    Tab1

    앞의 예제와 달리 탭 리스트를 동적으로 설정할 수 있으므로 관련 마크업은 생략할 수 있습니다. 

    <ul id="tab_2" class="tab tab-top"></ul>
<div id="tab_2_contents" style="background: #dcdcdc;">Tab1</div>

    nodes 옵션을 사용하여 탭 리스트 초기 값을 설정할 수 있습니다.
    그리고 tab2Run 함수와 같이 탭을 동적으로 조작할 수 있는 기능을 사용할 수 있습니다. 

    tab_2 = tab("#tab_2", {
	nodes: [
		{ text: "Tab1" },
		{ text: "Tab2" },
		{ text: "Tab3" }
	],
	event: {
		change: function(data) {
			$("#tab_2_contents").html(data.text);
		}
	},
	tpl: {
		node: "<li><a href='#'><!= text !></a></li>"
	}
});

function tab2Run(type) {
	switch(type) {
		case 1:
			tab_2_count += 1;
			tab_2.append({ text: "Tab" + tab_2_count });
			
			alert("탭을 마지막 위치에 추가합니다.");
			break;
		case 2:
			tab_2_count += 1;
			tab_2.prepend({ text: "Tab" + tab_2_count });
			
			alert("탭을 처음 위치에 추가합니다.");
			break;
		case 3:
			tab_2_count += 1;
			tab_2.insert(2, { text: "Tab" + tab_2_count });

			alert("2번 탭 위치에 추가합니다.");
			break;
		case 4:
			tab_2.remove(0);

			alert("0번 탭을 삭제합니다.");
			break;
		case 5:
			tab_2.move(0, 2);
			
			alert("0번 탭을 2번 탭 위치로 변경합니다.");
			break;
	}
}
