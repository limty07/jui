Layout

Default

Layout can split the screen into areas and each area will be resizable.

Left
Center
Right

Each area features an option to determine the basic/minimum/maximum size and a resizing option. 

jui.ready([ "ui.layout" ], function(layout) {
	layout_1 = layout("#container", {
		width: "auto",
		height: 300,
		left: {
			size: 100,
			min: 100,
			max: 300,
			resize: true
		},
		right: {
			size: 100,
			min: 100,
			max: 300,
			resize: true
		}
	});
});
<div id="container">
	<div class="left"><div class="title">Left</div></div>
	<div class="center"><div class="title">Center</div></div>
	<div class="right"><div class="title">Right</div></div>
</div>
