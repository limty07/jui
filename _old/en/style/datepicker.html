For reference, in this example, the number of basic columns is 3, not 7. 

<div class="datepicker datepicker-large">
    <div class="head">
        <div class="prev"></div>
        <div class="title">2013</div>
        <div class="next"></div>
    </div>
    <table class="body">
        <tr>
            <td>2004</td><td>2005</td><td>2006</td>
        </tr>
        <tr>
            <td>2007</td><td class="active">2008</td><td>2009</td>
        </tr>
        <tr>
            <td>2010</td><td>2011</td><td>2012</td>
        </tr>
        <tr>
            <td class="now">2013</td><td>2014</td><td>2015</td>
        </tr>
    </table>
</div>
